        e.preventDefault();
        sessionStorage.removeItem("access_token");
        sessionStorage.removeItem("roles");
        window.isLoggedIn = false;
        window.open("/", "_self")
    }

    addNewClick(e) {
        e.preventDefault();

                    $(".main-sidebar").css({ left: '0px' })

                    $(".main-sidebar").css({ left: '-230px' })
